  <!-- JS: Connect to Make Webhook -->
  <script>
    const form = document.getElementById('jdForm');
    const loading = document.getElementById('loading');
    const result = document.getElementById('result');

    form.addEventListener('submit', async (e) => {
      e.preventDefault();
      loading.classList.remove('hidden');
      result.classList.add('hidden');

      const data = {
        company: form.company.value,
        role: form.role.value,
        type: form.type.value,
        location: form.location.value,
        experience: form.experience.value,
        responsibilities: form.responsibilities.value,
        requirements: form.requirements.value,
        email: form.email.value
      };

      try {
        const res = await fetch('PASTE_YOUR_MAKE_WEBHOOK_URL_HERE', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(data)
        });
        const text = await res.text();
        result.innerText = text;
      } catch (err) {
        result.innerText = '‚ùå Error generating job description. Please try again.';
      }

      loading.classList.add('hidden');
      result.classList.remove('hidden');
    });
  </script>
